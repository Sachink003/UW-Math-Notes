\section{Linear Transformations}
\subsection{Rank and Nullity}
\begin{thm}
    Let $V$ and $W$ be vector space over $\F$, and let $L: V \to W$ be a linear map. Then
    \begin{enumerate}
        \item $L(\vec{0}) = \vec{0}$
        \item $\text{range}(L)$ is a subspace of $W$
        \item $\ker(L)$ is a subspace of $V$
    \end{enumerate}
\end{thm}
\begin{thm}[Rank-Nullity Theorem]
    Let $V$ and $W$ be vector spaces over $\mathbb{F}$ with $\dim(V) = n$. Let $L: V \to W$ be a linear map. 
    $$\text{rank}(L) + \text{nullity}(L) = n$$
\end{thm}
\subsection{Linear Maps as Matrices}
\begin{thm}
    Let $V$ be an $n$-dimensional vector space with ordered basis $\mathcal{B}$. Let $W$ be an $m$-dimensional vector space with ordered basis $\mathcal{C}$. Then, for every linear map 
    $L: V \to W$, there exists $A \in M_{m \times n}(\F)$ such that $[L(\vec{v})]_{\mathcal{C}} = A[\vec{v}]_{\mathcal{B}}$ for all $\vec{v} \in V$, which we denote $_{\mathcal{C}}[L]_{\mathcal{B}}$ is given by 
    \begin{align*}
      _{\mathcal{C}}[L]_{\mathcal{B}} = \big[[L(\vec{b}_1)]_{\mathcal{C}} \dots [L(\vec{b}_n)]_{\mathcal{C}}\big]
    \end{align*}
    % $$_{\mathcal{C}}[L]_{\mathcal{B}} = \right[[L(\vec{b}_1)]_{\mathcal{C}} \dots [L(\vec{b}_n)]_{\mathcal{C}}\left]$$
\end{thm}
